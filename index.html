                        await signInAnonymously(auth);

                    } catch (err) { 

                        console.error("Firebase Init Error", err);

                    }


                    onAuthStateChanged(auth, (user) => {

                        if (user) {

                            startSyncing();

                            checkSession();

                        }

                    });

                });


                const startSyncing = () => {

                    // 同步任务

                    const tasksRef = collection(db, 'artifacts', appId, 'public', 'data', 'tasks');

                    onSnapshot(tasksRef, (snap) => {

                        tasks.value = snap.docs.map(d => ({ id: d.id, ...d.data() }));

                        refreshIcons();

                    }, (err) => console.error("Tasks Sync Error:", err));


                    // 同步实习生

                    const internsRef = collection(db, 'artifacts', appId, 'public', 'data', 'interns');

                    onSnapshot(internsRef, (snap) => {

                        interns.value = snap.docs.map(d => ({ id: d.id, ...d.data() }));

                        refreshIcons();

                    }, (err) => console.error("Interns Sync Error:", err));

                };


                const checkSession = () => {

                    const saved = localStorage.getItem('vineyard_session_v4');

                    if (saved) {

                        const data = JSON.parse(saved);

                        currentUser.value = data;

                        view.value = data.role;

                        if (data.role === 'intern') syncMyProgress(data.id);

                    }

                    loading.value = false;

                    refreshIcons();

                };


                const syncMyProgress = (internId) => {

                    const internDoc = doc(db, 'artifacts', appId, 'public', 'data', 'interns', internId);

                    onSnapshot(internDoc, (snap) => {

                        if (snap.exists()) {

                            const finished = snap.data().finished || [];

                            const map = {};

                            finished.forEach(id => map[id] = true);

                            taskStatuses.value = map;

                        }

                    });

                };


                const handleLogin = async () => {

                    if (targetRole.value === 'admin') {

                        if (loginData.value.pass === '666') {

                            currentUser.value = { role: 'admin', name: 'Admin' };

                            view.value = 'admin';

                            localStorage.setItem('vineyard_session_v4', JSON.stringify(currentUser.value));

                            loginModal.value = false;

                        } else alert("密码错误");

                    } else {

                        const name = loginData.value.name.trim();

                        if (!name) return;

                        const internDoc = doc(db, 'artifacts', appId, 'public', 'data', 'interns', name);

                        await setDoc(internDoc, { name: name, lastSeen: Date.now() }, { merge: true });

                        currentUser.value = { role: 'intern', name: name, id: name };

                        view.value = 'intern';

                        localStorage.setItem('vineyard_session_v4', JSON.stringify(currentUser.value));

                        loginModal.value = false;

                        syncMyProgress(name);

                    }

                    refreshIcons();

                };


                const toggleTaskStatus = async (task) => {

                    const next = !taskStatuses.value[task.id];

                    taskStatuses.value[task.id] = next;

                    const finishedIds = Object.keys(taskStatuses.value).filter(id => taskStatuses.value[id]);

                    const internDoc = doc(db, 'artifacts', appId, 'public', 'data', 'interns', currentUser.value.id);

                    await updateDoc(internDoc, { finished: finishedIds, lastSeen: Date.now() });

                };


                const addTask = async () => {

                    if (!newTask.value.title) return;

                    const tasksCol = collection(db, 'artifacts', appId, 'public', 'data', 'tasks');

                    await addDoc(tasksCol, { ...newTask.value, createdAt: Date.now() });

                    newTask.value = { title: '', category: '教学部', description: '' };

                    showAddTask.value = false;

                };


                const deleteTask = async (id) => {

                    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'tasks', id));

                };


                const logout = () => {

                    localStorage.removeItem('vineyard_session_v4');

                    view.value = 'login';

                    currentUser.value = null;

                    refreshIcons();

                };


                const filteredTasks = computed(() => {

                    if (activeCat.value === 'All') return tasks.value;

                    return tasks.value.filter(t => t.category === activeCat.value);

                });


                const internProgress = computed(() => {

                    if (!tasks.value.length) return 0;

                    const doneCount = Object.values(taskStatuses.value).filter(Boolean).length;

                    return Math.round((doneCount / tasks.value.length) * 100);

                });


                const getInternPercent = (id) => {

                    const i = interns.value.find(x => x.id === id);

                    if (!i || !tasks.value.length) return 0;

                    return Math.round(((i.finished || []).length / tasks.value.length) * 100);

                };


                const getTagStyle = (cat) => {

                    if (cat === '教学部') return 'bg-purple-100 text-[#534AB7] border-purple-200';

                    if (cat === '教务部') return 'bg-green-100 text-[#1D9E75] border-green-200';

                    if (cat === '行政部') return 'bg-orange-100 text-orange-600 border-orange-200';

                    return 'bg-gray-100 text-gray-400 border-gray-200';

                };


                const formatDate = (ts) => {

                    if (!ts) return '从未';

                    return new Date(ts).toLocaleString();

                };


                return {

                    loading, view, adminTab, currentUser, loginModal, targetRole, loginData,

                    tasks, interns, taskStatuses, selectedTask, showAddTask, newTask,

                    categories, activeCat, filteredTasks, internProgress,

                    showRoleModal: (r) => { targetRole.value = r; loginModal.value = true; refreshIcons(); },

                    handleLogin, logout, toggleTaskStatus, addTask, deleteTask, getInternPercent, getTagStyle, formatDate

                };

            }

        }).mount('#app');

    </script>

</body>

</html>
